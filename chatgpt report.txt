(ns climate.core)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; DATA DICTIONARY
;;
;; year        integer
;; temp        mean annual temperature (°C)
;; rain        annual rainfall (mm)
;;
;; Dataset is simulated climate data
;; similar to typical station summaries.
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(def climate-data
  [{:year 2015 :temp 16.2 :rain 620}
   {:year 2016 :temp 16.5 :rain 590}
   {:year 2017 :temp 16.7 :rain 610}
   {:year 2018 :temp 17.0 :rain 570}
   {:year 2019 :temp 17.3 :rain 540}
   {:year 2020 :temp 17.1 :rain 650}
   {:year 2021 :temp 17.4 :rain 560}
   {:year 2022 :temp 17.6 :rain 520}
   {:year 2023 :temp 17.8 :rain 500}])

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; BASIC STATISTICS
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn average [xs]
  (/ (reduce + xs) (count xs)))

(defn mean-temperature [data]
  (average (map :temp data)))

(defn mean-rainfall [data]
  (average (map :rain data)))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; TREND ESTIMATION
;; simple linear trend per year
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn slope [xs ys]
  (let [n (count xs)
        sumx (reduce + xs)
        sumy (reduce + ys)
        sumxy (reduce + (map * xs ys))
        sumx2 (reduce + (map #(* % %) xs))]
    (/ (- (* n sumxy) (* sumx sumy))
       (- (* n sumx2) (* sumx sumx)))))

(defn temperature-trend [data]
  (let [years (map :year data)
        temps (map :temp data)]
    (slope years temps)))

(defn rainfall-trend [data]
  (let [years (map :year data)
        rains (map :rain data)]
    (slope years rains)))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; DROUGHT DETECTION
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn drought-years [data threshold]
  (filter #(<= (:rain %) threshold) data))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; REPORT
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn report [data]

  (println "CLIMATE REPORT")
  (println "--------------")

  (println "Average temperature:"
           (mean-temperature data) "C")

  (println "Average rainfall:"
           (mean-rainfall data) "mm")

  (println "Temperature trend (deg/year):"
           (temperature-trend data))

  (println "Rainfall trend (mm/year):"
           (rainfall-trend data))

  (println "\nPossible drought years (<550mm):")

  (doseq [y (drought-years data 550)]
    (println (:year y) (:rain y) "mm"))
)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn -main []
  (report climate-data))

(ns climate.core)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; SYNTHETIC CLIMATE PROJECT DEMO
;;
;; DATA DICTIONARY
;;
;; Monthly record:
;;   :year        integer
;;   :month       integer 1..12
;;   :temp        mean monthly temperature in deg C
;;   :rain        monthly rainfall in mm
;;   :humidity    mean monthly relative humidity in %
;;
;; This program:
;;   1. Generates synthetic monthly climate data
;;   2. Aggregates to annual summaries
;;   3. Detects hot/dry years
;;   4. Prints a simple climate report
;;
;; No external libraries required.
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; BASIC MATH HELPERS
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(def pi 3.141592653589793)

(defn sqr [x]
  (* x x))

(defn average [xs]
  (if (seq xs)
    (/ (reduce + xs) (double (count xs)))
    0.0))

(defn clamp [x lo hi]
  (max lo (min hi x)))

(defn round2 [x]
  (/ (Math/round (* x 100.0)) 100.0))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; SIMPLE DETERMINISTIC NOISE
;;
;; We avoid randomness so the same run always gives the same results.
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn synthetic-noise
  "Small deterministic pseudo-noise based on year and month."
  [year month scale]
  (* scale
     (+ (* 0.60 (Math/sin (+ (* year 0.37) (* month 1.91))))
        (* 0.40 (Math/cos (+ (* year 0.11) (* month 2.73)))))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; SEASONAL CLIMATE MODEL
;;
;; Southern Hemisphere style:
;; warmest around Jan/Feb, coolest around Jul
;; rainfall pattern is generic synthetic, not tied to a real station
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn seasonal-phase
  [month]
  ;; Shift so month 1 is near summer peak
  (/ (* 2.0 pi (- month 1)) 12.0))

(defn synthetic-temperature
  [year month]
  (let [phase (seasonal-phase month)
        ;; long-term warming trend
        trend (* 0.035 (- year 2000))
        seasonal (* 7.5 (Math/cos phase))
        variability (synthetic-noise year month 1.2)
        base 18.0]
    (round2 (+ base trend seasonal variability))))

(defn synthetic-rainfall
  [year month]
  (let [phase (seasonal-phase month)
        ;; mild annual drying trend
        trend (* -0.9 (- year 2000))
        ;; rainfall peaks in cooler months for this synthetic station
        seasonal (+ 55.0 (* 22.0 (- 1.0 (Math/cos phase))))
        variability (synthetic-noise (+ year 13) (+ month 7) 12.0)]
    (-> (+ seasonal trend variability)
        (clamp 5.0 180.0)
        round2)))

(defn synthetic-humidity
  [year month temp rain]
  (let [raw (+ 62.0
               (* 0.10 rain)
               (* -0.90 (- temp 18.0))
               (synthetic-noise (+ year 5) (+ month 11) 4.0))]
    (-> raw
        (clamp 25.0 98.0)
        round2)))

(defn make-record
  [year month]
  (let [temp (synthetic-temperature year month)
        rain (synthetic-rainfall year month)
        humidity (synthetic-humidity year month temp rain)]
    {:year year
     :month month
     :temp temp
     :rain rain
     :humidity humidity}))

(defn generate-climate-data
  [start-year end-year]
  (vec
   (for [year (range start-year (inc end-year))
         month (range 1 13)]
     (make-record year month))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; AGGREGATION
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn annual-summary
  [records]
  (let [year (:year (first records))
        temps (map :temp records)
        rains (map :rain records)
        hums  (map :humidity records)]
    {:year year
     :mean-temp (round2 (average temps))
     :total-rain (round2 (reduce + rains))
     :mean-humidity (round2 (average hums))
     :max-temp (apply max temps)
     :min-temp (apply min temps)}))

(defn annual-summaries
  [data]
  (->> data
       (group-by :year)
       (sort-by key)
       (map (fn [[_ recs]] (annual-summary recs)))
       vec))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; TREND ANALYSIS
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn slope
  [xs ys]
  (let [n (count xs)
        sumx (reduce + xs)
        sumy (reduce + ys)
        sumxy (reduce + (map * xs ys))
        sumx2 (reduce + (map sqr xs))
        denom (- (* n sumx2) (* sumx sumx))]
    (if (zero? denom)
      0.0
      (/ (- (* n sumxy) (* sumx sumy))
         (double denom)))))

(defn temp-trend-per-year
  [annual]
  (let [years (map :year annual)
        temps (map :mean-temp annual)]
    (round2 (slope years temps))))

(defn rain-trend-per-year
  [annual]
  (let [years (map :year annual)
        rains (map :total-rain annual)]
    (round2 (slope years rains))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; EXTREME / ALERT LOGIC
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn hot-year?
  [annual-rec threshold]
  (>= (:mean-temp annual-rec) threshold))

(defn dry-year?
  [annual-rec threshold]
  (<= (:total-rain annual-rec) threshold))

(defn hot-dry-years
  [annual temp-threshold rain-threshold]
  (filter #(and (hot-year? % temp-threshold)
                (dry-year? % rain-threshold))
          annual))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; PRINTING
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn print-month-sample
  [data n]
  (println "SAMPLE MONTHLY SYNTHETIC DATA")
  (println "----------------------------")
  (doseq [r (take n data)]
    (println
     (format "%d-%02d  temp=%5.2f C  rain=%6.2f mm  humidity=%5.2f %%"
             (:year r) (:month r) (:temp r) (:rain r) (:humidity r))))
  (println))

(defn print-annual-table
  [annual]
  (println "ANNUAL SUMMARIES")
  (println "----------------")
  (println "Year   MeanTemp(C)  TotalRain(mm)  MeanHumidity(%)  MinTemp  MaxTemp")
  (doseq [r annual]
    (println
     (format "%4d   %10.2f   %13.2f   %15.2f   %7.2f  %7.2f"
             (:year r)
             (:mean-temp r)
             (:total-rain r)
             (:mean-humidity r)
             (:min-temp r)
             (:max-temp r))))
  (println))

(defn print-extremes
  [annual]
  (let [hdy (hot-dry-years annual 19.0 700.0)]
    (println "HOT/DRY YEARS")
    (println "-------------")
    (if (seq hdy)
      (doseq [r hdy]
        (println
         (format "%d  mean-temp=%.2f C  total-rain=%.2f mm"
                 (:year r) (:mean-temp r) (:total-rain r))))
      (println "None under current thresholds."))
    (println)))

(defn print-trends
  [annual]
  (println "TRENDS")
  (println "------")
  (println (format "Temperature trend: %.2f deg C per year"
                   (temp-trend-per-year annual)))
  (println (format "Rainfall trend:    %.2f mm per year"
                   (rain-trend-per-year annual)))
  (println))

(defn report
  [data]
  (let [annual (annual-summaries data)]
    (print-month-sample data 12)
    (print-annual-table annual)
    (print-trends annual)
    (print-extremes annual)))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; MAIN
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn -main
  [& _args]
  (let [data (generate-climate-data 2015 2025)]
    (report data)))

(ns climate.core)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; DATA DICTIONARY
;;
;; year
;; month
;; temp      temperature C
;; rain      rainfall mm
;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(def pi 3.1415926535)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; SYNTHETIC CLIMATE GENERATOR
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn seasonal [m]
  (Math/cos (/ (* 2 pi (- m 1)) 12)))

(defn temp-model [year month]

  (let [trend (* 0.03 (- year 2000))
        seasonal (* 8 (seasonal month))
        noise (* 1.5 (Math/sin (+ year month)))]
    (+ 17 trend seasonal noise)))

(defn rain-model [year month]

  (let [trend (* -1 (- year 2000))
        seasonal (+ 60 (* 25 (- 1 (seasonal month))))
        noise (* 10 (Math/cos (+ year (* 2 month))))]
    (max 5 (+ seasonal trend noise))))

(defn record [year month]

  {:year year
   :month month
   :temp (temp-model year month)
   :rain (rain-model year month)})

(defn generate-data [start end]

  (vec
   (for [y (range start (inc end))
         m (range 1 13)]
     (record y m))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; BASIC STATS
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn mean [xs]
  (/ (reduce + xs) (count xs)))

(defn stddev [xs]

  (let [m (mean xs)
        v (mean (map #(Math/pow (- % m) 2) xs))]
    (Math/sqrt v)))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; DROUGHT INDEX (SPI style)
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn drought-index [rain]

  (let [m (mean rain)
        s (stddev rain)]
    (map #(/ (- % m) s) rain)))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; FLOOD ALERT
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn flood-events [data threshold]

  (filter #(> (:rain %) threshold) data))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; CSV EXPORT
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn write-csv [file data]

  (spit file
        (apply str
               "year,month,temp,rain\n"
               (for [r data]
                 (format "%d,%d,%.2f,%.2f\n"
                         (:year r)
                         (:month r)
                         (:temp r)
                         (:rain r))))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; ASCII RAINFALL CHART
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn rain-chart [data]

  (println "\nRainfall Chart")
  (println "--------------")

  (doseq [r (take 24 data)]

    (let [bars (int (/ (:rain r) 5))]
      (println (:year r) "-" (:month r)
               (apply str (repeat bars "*"))))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; REPORT
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn report [data]

  (println "\nSynthetic Climate Report")
  (println "-----------------------")

  (let [temps (map :temp data)
        rains (map :rain data)
        di (drought-index rains)
        floods (flood-events data 120)]

    (println "Average Temp:" (mean temps))
    (println "Average Rain:" (mean rains))

    (println "\nDrought Index Sample")
    (println (take 12 di))

    (println "\nFlood Events")
    (doseq [f floods]
      (println f))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; MAIN
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn -main []

  (let [data (generate-data 2015 2025)]

    (report data)

    (rain-chart data)

    (write-csv "climate_data.csv" data)

    (println "\nCSV file written: climate_data.csv")))

(ns climate.core)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; SYNTHETIC CLIMATE / HYDROLOGY / CLASSIFICATION DEMO
;;
;; DATA DICTIONARY
;;
;; Monthly climate record:
;;   :year            integer
;;   :month           integer 1..12
;;   :enso            ENSO-like oscillation index
;;   :temp            mean monthly temperature (deg C)
;;   :rain            monthly rainfall (mm)
;;   :humidity        mean monthly relative humidity (%)
;;   :pressure        synthetic mean sea-level pressure (hPa)
;;   :wind            synthetic mean wind speed (km/h)
;;   :cyclone?        boolean, synthetic intense storm month
;;   :drought-index   z-score style rainfall anomaly
;;   :flow            synthetic river flow index
;;   :flood-risk      :low / :moderate / :high
;;
;; This demo:
;;   1. Generates synthetic climate data
;;   2. Adds an ENSO-like oscillation
;;   3. Injects cyclone-enhanced rainfall months
;;   4. Computes drought index
;;   5. Simulates river response
;;   6. Trains a tiny linear classifier
;;   7. Exports CSV
;;   8. Prints reports and ASCII charts
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(def pi 3.141592653589793)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; BASIC HELPERS
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn sqr [x]
  (* x x))

(defn average [xs]
  (if (seq xs)
    (/ (reduce + xs) (double (count xs)))
    0.0))

(defn clamp [x lo hi]
  (max lo (min hi x)))

(defn round2 [x]
  (/ (Math/round (* x 100.0)) 100.0))

(defn abs [x]
  (Math/abs (double x)))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; DETERMINISTIC SYNTHETIC NOISE
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn noise
  [a b scale]
  (* scale
     (+ (* 0.52 (Math/sin (+ (* a 0.37) (* b 1.91))))
        (* 0.33 (Math/cos (+ (* a 0.17) (* b 2.47))))
        (* 0.15 (Math/sin (+ (* a 0.07) (* b 4.11)))))))

(defn seasonal-phase
  [month]
  (/ (* 2.0 pi (- month 1)) 12.0))

(defn sh-summer-cos
  [month]
  ;; 1 near January peak, -1 near July
  (Math/cos (seasonal-phase month)))

(defn sh-winter-factor
  [month]
  ;; 0 near summer, 1 near winter peak
  (/ (+ 1.0 (- (sh-summer-cos month))) 2.0))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; ENSO-LIKE INDEX
;;
;; Slow oscillation across years/months.
;; Positive => warmer/drier signal in this synthetic model.
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn enso-index
  [year month]
  (let [t (+ (* (- year 2000) 12.0) month)
        low-freq (* 1.25 (Math/sin (/ t 19.0)))
        med-freq (* 0.55 (Math/cos (/ t 9.0)))
        perturb (noise (+ year 40) (+ month 90) 0.18)]
    (round2 (+ low-freq med-freq perturb))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; CYCLONE / SEVERE STORM SIGNAL
;;
;; More likely in warmer months and when moisture is high.
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn cyclone-month?
  [year month enso]
  (let [summerish (max 0.0 (sh-summer-cos month))
        score (+ (* 1.25 summerish)
                 (* 0.18 (- 1.2 enso))
                 (noise (+ year 70) (+ month 33) 0.9))]
    (> score 1.25)))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; SYNTHETIC CLIMATE VARIABLES
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn synthetic-temp
  [year month enso cyclone?]
  (let [base 18.3
        warming (* 0.032 (- year 2000))
        seasonal (* 7.8 (sh-summer-cos month))
        enso-effect (* 0.75 enso)
        cyclone-cooling (if cyclone? -0.6 0.0)
        perturb (noise year month 1.1)]
    (-> (+ base warming seasonal enso-effect cyclone-cooling perturb)
        round2)))

(defn synthetic-rain
  [year month enso cyclone?]
  (let [base 40.0
        winter-boost (* 48.0 (sh-winter-factor month))
        drying (* -0.65 (- year 2000))
        enso-effect (* -9.0 enso)
        cyclone-burst (if cyclone? (+ 85.0 (abs (noise (+ year 5) (+ month 7) 45.0))) 0.0)
        perturb (noise (+ year 11) (+ month 19) 12.0)]
    (-> (+ base winter-boost drying enso-effect cyclone-burst perturb)
        (clamp 2.0 260.0)
        round2)))

(defn synthetic-humidity
  [year month temp rain cyclone?]
  (let [raw (+ 56.0
               (* 0.15 rain)
               (* -0.95 (- temp 18.0))
               (if cyclone? 6.0 0.0)
               (noise (+ year 3) (+ month 21) 4.5))]
    (-> raw
        (clamp 20.0 99.0)
        round2)))

(defn synthetic-pressure
  [year month enso cyclone?]
  (let [base 1014.0
        seasonal (* 3.4 (sh-winter-factor month))
        enso-effect (* 0.8 enso)
        cyclone-drop (if cyclone? (+ -11.0 (noise (+ year 8) (+ month 1) 2.0)) 0.0)
        perturb (noise (+ year 17) (+ month 23) 1.8)]
    (-> (+ base seasonal enso-effect cyclone-drop perturb)
        round2)))

(defn synthetic-wind
  [year month cyclone?]
  (let [base 14.0
        winter (* 4.0 (sh-winter-factor month))
        cyclone-boost (if cyclone? (+ 24.0 (abs (noise (+ year 31) (+ month 29) 8.0))) 0.0)
        perturb (noise (+ year 13) (+ month 27) 2.2)]
    (-> (+ base winter cyclone-boost perturb)
        (clamp 2.0 85.0)
        round2)))

(defn make-base-record
  [year month]
  (let [enso (enso-index year month)
        cyclone? (cyclone-month? year month enso)
        temp (synthetic-temp year month enso cyclone?)
        rain (synthetic-rain year month enso cyclone?)
        humidity (synthetic-humidity year month temp rain cyclone?)
        pressure (synthetic-pressure year month enso cyclone?)
        wind (synthetic-wind year month cyclone?)]
    {:year year
     :month month
     :enso enso
     :temp temp
     :rain rain
     :humidity humidity
     :pressure pressure
     :wind wind
     :cyclone? cyclone?}))

(defn generate-data
  [start-year end-year]
  (vec
   (for [y (range start-year (inc end-year))
         m (range 1 13)]
     (make-base-record y m))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; DROUGHT INDEX
;;
;; Simple z-score style monthly anomaly over whole series.
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn stddev
  [xs]
  (let [m (average xs)
        v (average (map #(sqr (- % m)) xs))]
    (Math/sqrt v)))

(defn add-drought-index
  [data]
  (let [rains (map :rain data)
        m (average rains)
        s (stddev rains)
        denom (if (< s 1.0e-9) 1.0 s)]
    (vec
     (map (fn [r]
            (assoc r :drought-index
                   (round2 (/ (- (:rain r) m) denom))))
          data))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; HYDROLOGY / RIVER RESPONSE
;;
;; Very simple monthly storage + runoff model.
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn evap-estimate
  [temp humidity]
  ;; synthetic PET-like index
  (-> (+ 18.0
         (* 1.25 (max 0.0 (- temp 10.0)))
         (* -0.10 humidity))
      (clamp 5.0 85.0)))

(defn flow-category
  [flow]
  (cond
    (>= flow 165.0) :high
    (>= flow 100.0) :moderate
    :else :low))

(defn simulate-river
  [data]
  (loop [remaining data
         soil 80.0
         out []]
    (if (empty? remaining)
      (vec out)
      (let [r (first remaining)
            pet (evap-estimate (:temp r) (:humidity r))
            soil-next (clamp (+ soil (* 0.55 (:rain r)) (* -0.70 pet)) 10.0 220.0)
            excess (max 0.0 (- soil-next 130.0))
            cyclone-runoff (if (:cyclone? r) 28.0 0.0)
            flow (+ (* 0.38 (:rain r))
                    (* 0.46 excess)
                    cyclone-runoff
                    (* 0.25 (max 0.0 (- 95.0 pet))))
            flowv (round2 flow)]
        (recur (rest remaining)
               soil-next
               (conj out
                     (assoc r
                            :pet (round2 pet)
                            :soil (round2 soil-next)
                            :flow flowv
                            :flood-risk (flow-category flowv))))))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; LABELING FOR A TINY ML STYLE CLASSIFIER
;;
;; Target:
;;   0 = low flood risk
;;   1 = elevated flood/flood-prone month
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn label-target
  [r]
  (if (or (= (:flood-risk r) :high)
          (and (:cyclone? r) (> (:rain r) 120.0))
          (> (:flow r) 150.0))
    1.0
    0.0))

(defn feature-vector
  [r]
  ;; [bias, rain, humidity, wind, pressure anomaly, drought anomaly, cyclone]
  [1.0
   (/ (:rain r) 100.0)
   (/ (:humidity r) 100.0)
   (/ (:wind r) 50.0)
   (/ (- 1018.0 (:pressure r)) 20.0)
   (/ (- (:drought-index r)) 3.0)
   (if (:cyclone? r) 1.0 0.0)])

(defn dot
  [xs ys]
  (reduce + (map * xs ys)))

(defn sigmoid
  [z]
  (/ 1.0 (+ 1.0 (Math/exp (- z)))))

(defn predict-proba
  [weights x]
  (sigmoid (dot weights x)))

(defn train-logistic
  [dataset epochs learning-rate]
  (let [nfeat (count (feature-vector (first dataset)))]
    (loop [epoch 0
           w (vec (repeat nfeat 0.0))]
      (if (>= epoch epochs)
        w
        (let [w2
              (reduce
               (fn [weights r]
                 (let [x (feature-vector r)
                       y (label-target r)
                       p (predict-proba weights x)
                       err (- y p)]
                   (vec
                    (map-indexed
                     (fn [i wi]
                       (+ wi (* learning-rate err (nth x i))))
                     weights))))
               w
               dataset)]
          (recur (inc epoch) w2))))))

(defn classify
  [weights r]
  (let [p (predict-proba weights (feature-vector r))]
    (assoc r
           :ml-prob (round2 p)
           :ml-class (if (>= p 0.5) :alert :normal))))

(defn apply-classifier
  [weights dataset]
  (mapv #(classify weights %) dataset))

(defn confusion-metrics
  [dataset]
  (reduce
   (fn [acc r]
     (let [truth (label-target r)
           pred (if (= (:ml-class r) :alert) 1.0 0.0)]
       (cond
         (and (= truth 1.0) (= pred 1.0)) (update acc :tp inc)
         (and (= truth 0.0) (= pred 0.0)) (update acc :tn inc)
         (and (= truth 0.0) (= pred 1.0)) (update acc :fp inc)
         :else (update acc :fn inc))))
   {:tp 0 :tn 0 :fp 0 :fn 0}
   dataset))

(defn annual-summary
  [records]
  {:year (:year (first records))
   :mean-temp (round2 (average (map :temp records)))
   :total-rain (round2 (reduce + (map :rain records)))
   :mean-flow (round2 (average (map :flow records)))
   :cyclone-months (count (filter :cyclone? records))
   :high-flood-months (count (filter #(= (:flood-risk %) :high) records))
   :mean-enso (round2 (average (map :enso records)))})

(defn annual-summaries
  [dataset]
  (->> dataset
       (group-by :year)
       (sort-by key)
       (map (fn [[_ recs]] (annual-summary recs)))
       vec))

(defn slope
  [xs ys]
  (let [n (count xs)
        sumx (reduce + xs)
        sumy (reduce + ys)
        sumxy (reduce + (map * xs ys))
        sumx2 (reduce + (map sqr xs))
        denom (- (* n sumx2) (* sumx sumx))]
    (if (zero? denom)
      0.0
      (/ (- (* n sumxy) (* sumx sumy))
         (double denom)))))

(defn trend
  [annual k]
  (round2
   (slope (map :year annual)
          (map k annual))))

        recall (if (zero? (+ tp fn)) 0.0 (/ tp (+ tp fn)))]

    (println "------------------")

What this project does

This version is closer to a mini research platform than a simple demo.

It synthesizes a monthly climate history from 2010 to 2025, then:

    adds a slow ENSO-like oscillation

    creates cyclone-prone months

    computes a monthly drought index

    turns rainfall and atmospheric conditions into a simple river flow

    labels flood-prone months

    trains a tiny logistic-style classifier

    exports the results to CSV

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; GRID-BASED SYNTHETIC CLIMATE MAP DEMO
;;
;; DATA DICTIONARY
;;
;; Grid cell:
;;   :row         integer
;;   :col         integer
;;   :lat         latitude (deg)
;;   :lon         longitude (deg)
;;   :elev        elevation (m)
;;   :coastal?    boolean
;;
;; Monthly climate record:
;;   :year
;;   :month
;;   :row
;;   :col
;;   :lat
;;   :lon
;;   :elev
;;   :coastal?
;;   :temp        deg C
;;   :rain        mm
;;   :humidity    %
;;   :pressure    hPa
;;   :wind        km/h
;;   :drought     z-score style rainfall anomaly
;;   :flood-risk  :low/:moderate/:high
;;
;; This demo:
;;   1. Builds a synthetic lat/lon grid
;;   2. Assigns terrain and coastal effects
;;   3. Generates monthly climate fields
;;   4. Computes drought anomaly
;;   5. Classifies flood risk
;;   6. Prints ASCII maps
;;   7. Exports CSV
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(def pi 3.141592653589793)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; BASIC HELPERS
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn sqr [x]
  (* x x))

(defn average [xs]
  (if (seq xs)
    (/ (reduce + xs) (double (count xs)))
    0.0))

(defn clamp [x lo hi]
  (max lo (min hi x)))

(defn round2 [x]
  (/ (Math/round (* x 100.0)) 100.0))

(defn abs [x]
  (Math/abs (double x)))

(defn stddev [xs]
  (let [m (average xs)
        v (average (map #(sqr (- % m)) xs))]
    (Math/sqrt v)))

(defn seasonal-phase [month]
  (/ (* 2.0 pi (- month 1)) 12.0))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; SYNTHETIC NOISE
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn noise
  [a b c scale]
  (* scale
     (+ (* 0.45 (Math/sin (+ (* a 0.21) (* b 0.73) (* c 1.11))))
        (* 0.35 (Math/cos (+ (* a 0.17) (* b 0.29) (* c 1.73))))
        (* 0.20 (Math/sin (+ (* a 0.09) (* b 0.41) (* c 2.31)))))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; GRID DEFINITION
;;
;; Rough Australia-like box:
;; lat from -44 to -10
;; lon from 112 to 154
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(def nrows 10)
(def ncols 14)

(def min-lat -44.0)
(def max-lat -10.0)
(def min-lon 112.0)
(def max-lon 154.0)

(defn lat-step []
  (/ (- max-lat min-lat) (dec nrows)))

(defn lon-step []
  (/ (- max-lon min-lon) (dec ncols)))

(defn grid-lat [row]
  (+ min-lat (* row (lat-step))))

(defn grid-lon [col]
  (+ min-lon (* col (lon-step))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; TERRAIN / LANDSCAPE
;;
;; Synthetic topography:
;; - inland ridge
;; - southeast highlands
;; - tropical north lower
;; - coastal flag near edges
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn eastness [lon]
  (/ (- lon min-lon) (- max-lon min-lon)))

(defn northness [lat]
  (/ (- lat min-lat) (- max-lat min-lat)))

(defn coastal?
  [row col]
  (or (<= row 1)
      (>= row (- nrows 2))
      (<= col 1)
      (>= col (- ncols 2))))

(defn synthetic-elevation
  [lat lon]
  (let [ridge (* 550.0
                 (Math/exp (- (+ (/ (sqr (- lon 141.0)) 28.0)
                                  (/ (sqr (- lat -26.0)) 70.0)))))
        se-highlands (* 950.0
                        (Math/exp (- (+ (/ (sqr (- lon 147.0)) 18.0)
                                         (/ (sqr (- lat -36.0)) 12.0)))))
        nw-lowlands (* 80.0
                       (Math/exp (- (+ (/ (sqr (- lon 122.0)) 90.0)
                                        (/ (sqr (- lat -18.0)) 30.0)))))
        base (+ 40.0 ridge se-highlands nw-lowlands
                (noise lat lon 1.0 35.0))]
    (round2 (clamp base 0.0 1800.0))))

(defn make-cell
  [row col]
  (let [lat (grid-lat row)
        lon (grid-lon col)]
    {:row row
     :col col
     :lat (round2 lat)
     :lon (round2 lon)
     :elev (synthetic-elevation lat lon)
     :coastal? (coastal? row col)}))

(defn make-grid []
  (vec
   (for [row (range nrows)
         col (range ncols)]
     (make-cell row col))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; CLIMATE FIELD GENERATORS
;;
;; Features:
;; - warmer north
;; - cooler south
;; - cooler with elevation
;; - coast moderates temperature
;; - wetter coast and southeast
;; - winter rainfall south, summer rainfall north
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn summer-factor
  [month]
  ;; 1 around Jan, -1 around Jul
  (Math/cos (seasonal-phase month)))

(defn winter-factor
  [month]
  ;; 0 around Jan, 1 around Jul
  (/ (+ 1.0 (- (summer-factor month))) 2.0))

(defn tropical-summer-rain-factor
  [month]
  ;; 1 around Jan-Feb, 0 around Jul
  (/ (+ 1.0 (summer-factor month)) 2.0))

(defn enso-index
  [year month]
  (let [t (+ (* (- year 2000) 12.0) month)]
    (round2
     (+ (* 1.10 (Math/sin (/ t 17.0)))
        (* 0.45 (Math/cos (/ t 8.0)))
        (noise year month 3.0 0.12)))))

(defn temp-field
  [cell year month enso]
  (let [{:keys [lat lon elev coastal?]} cell
        north-warm (* 0.62 (- -10.0 lat))
        base (+ 10.5 north-warm)
        seasonal (* 6.8 (summer-factor month) (- 1.10 (northness lat)))
        tropical-seasonal (* 2.2 (summer-factor month) (northness lat))
        elevation-cooling (* -0.0065 elev)
        coastal-moderation (if coastal?
                             (* -0.9 (abs (summer-factor month)))
                             0.7)
        enso-effect (* 0.55 enso)
        perturb (noise lat lon (+ year month) 1.0)]
    (-> (+ base seasonal tropical-seasonal elevation-cooling
           coastal-moderation enso-effect perturb)
        round2)))

(defn rain-field
  [cell year month enso]
  (let [{:keys [lat lon elev coastal?]} cell
        north (northness lat)
        east (eastness lon)
        south (- 1.0 north)
        coast-boost (if coastal? 18.0 0.0)
        orographic (* 0.018 elev)
        tropical-summer (* 95.0 north (tropical-summer-rain-factor month))
        southern-winter (* 58.0 south east (winter-factor month))
        interior-dry (* -38.0 (- 1.0 east) (- 1.0 (if coastal? 1.0 0.0)))
        enso-effect (* -10.0 enso)
        trend (* -0.45 (- year 2000))
        base 18.0
        perturb (noise lat lon (+ year (* 2 month)) 10.0)]
    (-> (+ base coast-boost orographic tropical-summer southern-winter
           interior-dry enso-effect trend perturb)
        (clamp 1.0 320.0)
        round2)))

(defn humidity-field
  [cell temp rain month]
  (let [{:keys [coastal? elev lat]} cell
        raw (+ 38.0
               (* 0.23 rain)
               (* -1.05 (- temp 20.0))
               (* -0.004 elev)
               (if coastal? 8.0 0.0)
               (* 6.0 (northness lat))
               (noise lat temp month 3.5))]
    (-> raw
        (clamp 15.0 99.0)
        round2)))

(defn pressure-field
  [cell year month enso rain]
  (let [{:keys [lat lon elev]} cell
        raw (+ 1013.5
               (* 2.8 (winter-factor month))
               (* 0.65 enso)
               (* -0.012 rain)
               (* -0.0015 elev)
               (noise lat lon (+ year month 5) 1.8))]
    (round2 raw)))

(defn wind-field
  [cell month rain]
  (let [{:keys [lat lon coastal?]} cell
        raw (+ 10.0
               (* 5.5 (winter-factor month))
               (* 0.03 rain)
               (if coastal? 4.0 0.0)
               (noise lat lon (+ month 8) 2.2))]
    (-> raw
        (clamp 1.0 85.0)
        round2)))

(defn flood-risk
  [rain elev coastal?]
  (cond
    (or (> rain 180.0)
        (and coastal? (> rain 140.0))
        (and (> elev 700.0) (> rain 130.0))) :high
    (or (> rain 110.0)
        (and coastal? (> rain 95.0))) :moderate
    :else :low))

(defn monthly-record
  [cell year month]
  (let [enso (enso-index year month)
        temp (temp-field cell year month enso)
        rain (rain-field cell year month enso)
        humidity (humidity-field cell temp rain month)
        pressure (pressure-field cell year month enso rain)
        wind (wind-field cell month rain)]
    (assoc cell
           :year year
           :month month
           :enso enso
           :temp temp
           :rain rain
           :humidity humidity
           :pressure pressure
           :wind wind
           :flood-risk (flood-risk rain (:elev cell) (:coastal? cell)))))

(defn generate-dataset
  [start-year end-year]
  (let [grid (make-grid)]
    (vec
     (for [year (range start-year (inc end-year))
           month (range 1 13)
           cell grid]
       (monthly-record cell year month)))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; DROUGHT INDEX
;;
;; Per grid cell across time.
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn cell-key [r]
  [(:row r) (:col r)])

(defn add-drought-index
  [dataset]
  (let [by-cell (group-by cell-key dataset)]
    (vec
     (mapcat
      (fn [[_ recs]]
        (let [rains (map :rain recs)
              m (average rains)
              s (stddev rains)
              denom (if (< s 1.0e-9) 1.0 s)]
          (map (fn [r]
                 (assoc r :drought
                        (round2 (/ (- (:rain r) m) denom))))
               recs)))
      by-cell))))

(defn annual-cell-summary
  [recs]
  (let [floods (frequencies (map :flood-risk recs))]
    {:year (:year (first recs))
     :row (:row (first recs))
     :col (:col (first recs))
     :lat (:lat (first recs))
     :lon (:lon (first recs))
     :elev (:elev (first recs))
     :coastal? (:coastal? (first recs))
     :mean-temp (round2 (average (map :temp recs)))
     :total-rain (round2 (reduce + (map :rain recs)))
     :mean-humidity (round2 (average (map :humidity recs)))
     :mean-pressure (round2 (average (map :pressure recs)))
     :mean-wind (round2 (average (map :wind recs)))
     :mean-drought (round2 (average (map :drought recs)))
     :high-flood-months (get floods :high 0)
     :moderate-flood-months (get floods :moderate 0)}))

(defn annual-summaries
  [dataset year]
  (->> dataset
       (filter #(= (:year %) year))
       (group-by (fn [r] [(:row r) (:col r)]))
       (map (fn [[_ recs]] (annual-cell-summary recs)))
       (sort-by (juxt :row :col))
       vec))

    (< t 8.0) \A

    (< t 20.0) \D
    (< t 24.0) \E

(defn rain-symbol [r]
  (cond
    (< r 200.0) \.
    (< r 400.0) \:
    (< r 700.0) \-
    (< r 1000.0) \=
    (< r 1400.0) \+
    (< r 1800.0) \*
    :else \#))

(defn drought-symbol [d]
  (cond
    (< d -1.5) \D
    (< d -0.5) \d
    (< d 0.5) \.
    (< d 1.5) \w
    :else \W))

(defn flood-symbol [n]
  (cond
    (>= n 5) \H
    (>= n 2) \M
    :else \L))

(defn grid->rows
  [annual k]
  (for [row (range nrows)]
    (apply str
           (for [col (range ncols)]
             (let [r (first (filter #(and (= (:row %) row)
                                          (= (:col %) col))
                                    annual))]
               (k r))))))

(defn print-map
  [title annual symfn value-key legend]
  (println title)
  (println (apply str (repeat (count title) \-)))
  (doseq [line (grid->rows annual #(symfn (value-key %)))]
    (println line))
  (println legend)
  (println))

  {:mean-temp (round2 (average (map :mean-temp annual)))
   :mean-rain (round2 (average (map :total-rain annual)))

   :mean-pressure (round2 (average (map :mean-pressure annual)))

(defn print-extremes
  [annual]
  (let [hottest (apply max-key :mean-temp annual)
        wettest (apply max-key :total-rain annual)
        driest  (apply min-key :total-rain annual)
        highest (apply max-key :elev annual)]
    (println "EXTREMES")
    (println "--------")
    (println (format "Hottest cell: row=%d col=%d lat=%.2f lon=%.2f mean-temp=%.2f"
                     (:row hottest) (:col hottest) (:lat hottest) (:lon hottest) (:mean-temp hottest)))
    (println (format "Wettest cell: row=%d col=%d lat=%.2f lon=%.2f total-rain=%.2f"
                     (:row wettest) (:col wettest) (:lat wettest) (:lon wettest) (:total-rain wettest)))
    (println (format "Driest cell:  row=%d col=%d lat=%.2f lon=%.2f total-rain=%.2f"
                     (:row driest) (:col driest) (:lat driest) (:lon driest) (:total-rain driest)))
    (println (format "Highest cell: row=%d col=%d lat=%.2f lon=%.2f elev=%.2f"
                     (:row highest) (:col highest) (:lat highest) (:lon highest) (:elev highest)))
    (println)))

What this does

This project creates a 2D climate grid rather than a single station.

Each cell gets:

    a synthetic latitude and longitude

    synthetic terrain height

    coastal or inland behaviour

    monthly weather fields

    drought anomaly

    flood classification

Then it prints ASCII maps for 2025.

Critique

It is a solid demo architecture and a good step up from a single-station synthetic model, but as a climate map project it is still closer to a stylized teaching simulator than a physically grounded model.

The strongest parts are the overall structure. The code is readable, self-contained, and easy to extend. You separated grid generation, terrain, climate fields, drought logic, summaries, maps, and CSV export cleanly. That makes it suitable for the kind of iterative technical refinement you usually want. The data dictionary is also useful and the row/col plus lat/lon design is sensible.

What it does well:

    turns the problem into a genuine 2D grid rather than one time series

    includes terrain and coastal effects, which makes the output more interesting

    computes per-cell drought anomalies over time

    produces annual and monthly views

    exports CSV, which is important for later analysis in Python, Rust, or spreadsheets

    stays dependency-free, so it is easy to run

Where it is weak:
1. The “Australia-like” geography is only symbolic

The domain box looks roughly Australian, but the land-sea structure is not realistic. Every cell is treated as land, and “coastal” just means near the edge of the grid. That creates visually plausible output, but not geographically meaningful output.

Why this matters:

    inland dryness and coastal wetness become artifacts of your rules rather than consequences of geography

    Tasmania, Cape York, the Great Dividing Range, central deserts, and southern ocean exposure are not really represented

    edge cells are automatically advantaged as “coastal,” even where a real landmass would not be

A better approach would be:

    define a land mask

    define an ocean mask

    make coastality depend on adjacency to ocean cells

    optionally encode a rough Australian polygon

2. The elevation field is attractive but arbitrary

The elevation function creates nice hills, especially southeast highlands, but it is synthetic in a decorative sense rather than climatological. It gives topography, but not a proper orographic control on windward/leeward rainfall.

Current limitation:

    rainfall increases with elevation directly, but no moisture-transport direction exists

    no rain-shadow effect

    no distinction between east-facing and west-facing slopes

A better model would:

    define prevailing seasonal wind directions

    compute upslope enhancement along those directions

    reduce rainfall on lee sides

That would give you a more convincing Great Dividing Range style effect.
3. Temperature formulation has a likely scaling issue

This line is suspicious:

north-warm (* 0.62 (- -10.0 lat))

Because latitudes are negative, this means cells farther south may get a larger positive adjustment than intended, depending on sign interpretation. For example, at latitude -44, (- -10.0 -44.0) becomes 34, so the “north-warm” term becomes large and positive. That seems backwards.

You probably intended northern cells to be warmer and southern cells cooler. A safer expression would be something like:

(defn northness [lat]
  (/ (- lat min-lat) (- max-lat min-lat)))

and then use:

(* 12.0 (northness lat))

or

(* -0.62 (abs (+ lat 10.0)))

depending on your convention.

This is the single most important numerical issue in the current design.
4. Drought index is simple, but not really a drought metric

You compute a per-cell z-score over rainfall history. That is fine as a teaching anomaly index, but it is not really SPI in a climatological sense.

Limitations:

    uses a plain normal-style standardization

    does not fit a gamma distribution or transform probabilities

    does not support multiple accumulation scales such as 3-, 6-, or 12-month drought

    does not include evaporative demand, so it misses hot drought

For a better drought product:

    compute rolling 3- and 12-month rainfall

    standardize these separately

    optionally add a simple PET-based dryness index

That would align much better with hydrology and drought monitoring.
5. Flood-risk classification is too threshold-based

The flood-risk logic is easy to understand, but it is static. It depends mostly on rainfall, elevation, and coastality in a single month.

Why that is limited:

    flood risk in reality depends on antecedent moisture and storage

    high elevation does not automatically increase flood risk

    a 140 mm coastal month can be harmless in one catchment and extreme in another

    no runoff routing is present

A better system would:

    track soil moisture or storage per cell

    compute runoff as a function of rainfall minus infiltration

    route excess water downstream by elevation gradient

    define flood risk from accumulated routed flow rather than just local rainfall

That would be much more in line with the flood and runoff modelling you tend to care about.
6. Performance is acceptable now, but some code patterns will not scale

The repeated use of filter to find a matching cell for each row/col in map printing is fine for a 10x14 grid, but it is inefficient for larger grids.

For example, here:

(first (filter #(and (= (:row %) row)
                     (= (:col %) col))
               annual))

This is repeated a lot. For a bigger grid or longer time range, that becomes costly.

Better:

    pre-index by [row col]

    then use direct lookup

For example:

(let [idx (into {} (map (fn [r] [[(:row r) (:col r)] r]) annual))]
  ...)

That will make the map functions cleaner and faster.
7. Grid orientation may confuse interpretation

You print rows from 0 upward, which means the southernmost latitude may appear first, depending on how grid-lat is defined. That is not wrong, but it can invert the intuitive north-up map view.

You should decide explicitly:

    print north at the top, or

    print south at the top

For climate maps, north-at-top usually feels more natural. That means reversing row display if row 0 is the south.
8. Annual drought averaging is questionable

You compute :mean-drought from monthly drought anomalies and then map that annually. That can blur the actual dryness pattern.

Why:

    drought anomalies often matter as persistence or minima, not average monthly anomaly

    averaging can cancel severe dry months with wet months

Better annual summaries might include:

    minimum monthly drought anomaly

    number of dry months below a threshold

    3-month rolling minimum

    12-month annual standardized anomaly

Those would make the annual drought map more meaningful.
9. Rainfall map symbols use annual totals, monthly map uses monthly thresholds

That is okay, but the legend system is inconsistent enough that users can misread it. The same symbol family means different numeric scales in different maps.

That is not a bug, but it is a usability weakness. It would help to:

    standardize symbol bands where possible

    print whether the field is monthly or annual in the legend

    maybe show min/max values

10. No temporal persistence except drought calculation

Each month is largely computed from formulas independently, except drought being standardized across time. Real climate fields have memory.

Missing persistence:

    soil moisture memory

    seasonal lag

    oceanic lag

    multi-month wet/dry spells

    persistent anomalies

A simple improvement would be to evolve one or two state variables through time per cell, such as:

    soil moisture

    groundwater/storage

    slow background climate anomaly

That alone would make the time evolution look much more realistic.
Overall verdict

As a synthetic demo, it is good.
As a research-style prototype, it is promising but still highly stylized.
As a climate-hydrology modelling base, it needs three major upgrades:

    fix the latitude-temperature logic

    add true land/ocean and routing structure

    replace threshold flood logic with storage/runoff propagation

My rating

For clarity and extensibility: 8/10
For physical realism: 4/10
For usefulness as a base for your kind of weather/flood work: 7/10
Best next version

The strongest next step would be a grid model with:

    land/ocean mask

    corrected north-south temperature gradient

    soil moisture state

    downstream runoff routing

    basin outlets

    flood propagation map

    monthly animation frames

That would move it much closer to the kind of synthetic hydrology platform that suits your meteorology and flood-modelling interests.

(defn sqr [x]
  (* x x))

(defn average [xs]
  (if (seq xs)
    (/ (reduce + xs) (double (count xs)))
    0.0))

(defn clamp [x lo hi]
  (max lo (min hi x)))

(defn round2 [x]
  (/ (Math/round (* x 100.0)) 100.0))

(defn abs [x]
  (Math/abs (double x)))

(defn stddev [xs]
  (let [m (average xs)
        v (average (map #(sqr (- % m)) xs))]
    (Math/sqrt v)))

(defn seasonal-phase [month]
  (/ (* 2.0 pi (- month 1)) 12.0))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn noise
  [a b c scale]
  (* scale
     (+ (* 0.45 (Math/sin (+ (* a 0.21) (* b 0.73) (* c 1.11))))
        (* 0.35 (Math/cos (+ (* a 0.17) (* b 0.29) (* c 1.73))))
        (* 0.20 (Math/sin (+ (* a 0.09) (* b 0.41) (* c 2.31)))))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(def ncols 14)

(def min-lat -44.0)
(def max-lat -10.0)
(def min-lon 112.0)
(def max-lon 154.0)

(defn lat-step []
  (/ (- max-lat min-lat) (dec nrows)))

(defn lon-step []
  (/ (- max-lon min-lon) (dec ncols)))

(defn grid-lat [row]
  (+ min-lat (* row (lat-step))))

(defn grid-lon [col]
  (+ min-lon (* col (lon-step))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; TERRAIN / LANDSCAPE
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn eastness [lon]
  (/ (- lon min-lon) (- max-lon min-lon)))

(defn northness [lat]
  (/ (- lat min-lat) (- max-lat min-lat)))

(defn coastal?
  [row col]
  (or (<= row 1)
      (>= row (- nrows 2))
      (<= col 1)
      (>= col (- ncols 2))))

(defn synthetic-elevation
  [lat lon]
  (let [ridge (* 550.0
                 (Math/exp (- (+ (/ (sqr (- lon 141.0)) 28.0)
                                  (/ (sqr (- lat -26.0)) 70.0)))))
        se-highlands (* 950.0
                        (Math/exp (- (+ (/ (sqr (- lon 147.0)) 18.0)
                                         (/ (sqr (- lat -36.0)) 12.0)))))
        nw-lowlands (* 80.0
                       (Math/exp (- (+ (/ (sqr (- lon 122.0)) 90.0)
                                        (/ (sqr (- lat -18.0)) 30.0)))))
        base (+ 40.0 ridge se-highlands nw-lowlands
                (noise lat lon 1.0 35.0))]
    (round2 (clamp base 0.0 1800.0))))

(defn make-cell
  [row col]
  (let [lat (grid-lat row)
        lon (grid-lon col)]
    {:row row
     :col col
     :lat (round2 lat)
     :lon (round2 lon)
     :elev (synthetic-elevation lat lon)
     :coastal? (coastal? row col)}))

(defn make-grid []
  (vec
   (for [row (range nrows)
         col (range ncols)]
     (make-cell row col))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; CLIMATE FIELD GENERATORS
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn summer-factor
  [month]
  (Math/cos (seasonal-phase month)))

(defn winter-factor
  [month]
  (/ (+ 1.0 (- (summer-factor month))) 2.0))

(defn tropical-summer-rain-factor
  [month]
  (/ (+ 1.0 (summer-factor month)) 2.0))

(defn enso-index
  [year month]
  (let [t (+ (* (- year 2000) 12.0) month)]
    (round2
     (+ (* 1.10 (Math/sin (/ t 17.0)))
        (* 0.45 (Math/cos (/ t 8.0)))
        (noise year month 3.0 0.12)))))

(defn temp-field
  [cell year month enso]
  (let [{:keys [lat lon elev coastal?]} cell
        ;; corrected north-south temperature gradient
        north-warm (* 12.0 (northness lat))
        base (+ 8.5 north-warm)
        seasonal (* 6.8 (summer-factor month) (- 1.10 (northness lat)))
        tropical-seasonal (* 2.2 (summer-factor month) (northness lat))
        elevation-cooling (* -0.0065 elev)
        coastal-moderation (if coastal?
                             (* -0.9 (abs (summer-factor month)))
                             0.7)
        enso-effect (* 0.55 enso)
        perturb (noise lat lon (+ year month) 1.0)]
    (-> (+ base seasonal tropical-seasonal elevation-cooling
           coastal-moderation enso-effect perturb)
        round2)))

(defn rain-field
  [cell year month enso]
  (let [{:keys [lat lon elev coastal?]} cell
        north (northness lat)
        east (eastness lon)
        south (- 1.0 north)
        coast-boost (if coastal? 18.0 0.0)
        orographic (* 0.018 elev)
        tropical-summer (* 95.0 north (tropical-summer-rain-factor month))
        southern-winter (* 58.0 south east (winter-factor month))
        interior-dry (* -38.0 (- 1.0 east) (- 1.0 (if coastal? 1.0 0.0)))
        enso-effect (* -10.0 enso)
        trend (* -0.45 (- year 2000))
        base 18.0
        perturb (noise lat lon (+ year (* 2 month)) 10.0)]
    (-> (+ base coast-boost orographic tropical-summer southern-winter
           interior-dry enso-effect trend perturb)
        (clamp 1.0 320.0)
        round2)))

(defn humidity-field
  [cell temp rain month]
  (let [{:keys [coastal? elev lat]} cell
        raw (+ 38.0
               (* 0.23 rain)
               (* -1.05 (- temp 20.0))
               (* -0.004 elev)
               (if coastal? 8.0 0.0)
               (* 6.0 (northness lat))
               (noise lat temp month 3.5))]
    (-> raw
        (clamp 15.0 99.0)
        round2)))

(defn pressure-field
  [cell year month enso rain]
  (let [{:keys [lat lon elev]} cell
        raw (+ 1013.5
               (* 2.8 (winter-factor month))
               (* 0.65 enso)
               (* -0.012 rain)
               (* -0.0015 elev)
               (noise lat lon (+ year month 5) 1.8))]
    (round2 raw)))

(defn wind-field
  [cell month rain]
  (let [{:keys [lat lon coastal?]} cell
        raw (+ 10.0
               (* 5.5 (winter-factor month))
               (* 0.03 rain)
               (if coastal? 4.0 0.0)
               (noise lat lon (+ month 8) 2.2))]
    (-> raw
        (clamp 1.0 85.0)
        round2)))

(defn flood-risk
  [rain elev coastal?]
  (cond
    (or (> rain 180.0)
        (and coastal? (> rain 140.0))
        (and (> elev 700.0) (> rain 130.0))) :high
    (or (> rain 110.0)
        (and coastal? (> rain 95.0))) :moderate
    :else :low))

(defn monthly-record
  [cell year month]
  (let [enso (enso-index year month)
        temp (temp-field cell year month enso)
        rain (rain-field cell year month enso)
        humidity (humidity-field cell temp rain month)
        pressure (pressure-field cell year month enso rain)
        wind (wind-field cell month rain)]
    (assoc cell
           :year year
           :month month
           :enso enso
           :temp temp
           :rain rain
           :humidity humidity
           :pressure pressure
           :wind wind
           :flood-risk (flood-risk rain (:elev cell) (:coastal? cell)))))

(defn generate-dataset
  [start-year end-year]
  (let [grid (make-grid)]
    (vec
     (for [year (range start-year (inc end-year))
           month (range 1 13)
           cell grid]
       (monthly-record cell year month)))))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; DROUGHT INDEX
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn cell-key [r]
  [(:row r) (:col r)])

(defn add-drought-index
  [dataset]
  (let [by-cell (group-by cell-key dataset)]
    (->> by-cell
         (mapcat
          (fn [[_ recs]]
            (let [ordered (sort-by (juxt :year :month) recs)
                  rains (map :rain ordered)
                  m (average rains)
                  s (stddev rains)
                  denom (if (< s 1.0e-9) 1.0 s)]
              (map (fn [r]
                     (assoc r :drought
                            (round2 (/ (- (:rain r) m) denom))))
                   ordered))))
         vec)))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; ANNUAL GRID SUMMARY
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn annual-cell-summary
  [recs]
  (let [floods (frequencies (map :flood-risk recs))]
    {:year (:year (first recs))
     :row (:row (first recs))
     :col (:col (first recs))
     :lat (:lat (first recs))
     :lon (:lon (first recs))
     :elev (:elev (first recs))
     :coastal? (:coastal? (first recs))
     :mean-temp (round2 (average (map :temp recs)))
     :total-rain (round2 (reduce + (map :rain recs)))
     :mean-humidity (round2 (average (map :humidity recs)))
     :mean-pressure (round2 (average (map :pressure recs)))
     :mean-wind (round2 (average (map :wind recs)))
     :mean-drought (round2 (average (map :drought recs)))
     :high-flood-months (get floods :high 0)
     :moderate-flood-months (get floods :moderate 0)}))

(defn annual-summaries
  [dataset year]
  (->> dataset
       (filter #(= (:year %) year))
       (group-by (fn [r] [(:row r) (:col r)]))
       (map (fn [[_ recs]] (annual-cell-summary recs)))
       (sort-by (juxt :row :col))
       vec))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; ASCII MAP HELPERS
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn temp-symbol [t]
  (cond
    (< t 8.0) \A
    (< t 12.0) \B
    (< t 16.0) \C
    (< t 20.0) \D
    (< t 24.0) \E
    (< t 28.0) \F
    :else \G))

(defn rain-symbol [r]
  (cond
    (< r 200.0) \.
    (< r 400.0) \:
    (< r 700.0) \-
    (< r 1000.0) \=
    (< r 1400.0) \+
    (< r 1800.0) \*
    :else \#))

(defn drought-symbol [d]
  (cond
    (< d -1.5) \D
    (< d -0.5) \d
    (< d 0.5) \.
    (< d 1.5) \w
    :else \W))

(defn flood-symbol [n]
  (cond
    (>= n 5) \H
    (>= n 2) \M
    :else \L))

(defn indexed-by-cell
  [records]
  (into {} (map (fn [r] [[(:row r) (:col r)] r]) records)))

(defn grid->rows
  [annual k]
  (let [idx (indexed-by-cell annual)]
    ;; north at top
    (for [row (reverse (range nrows))]
      (apply str
             (for [col (range ncols)]
               (let [r (get idx [row col])]
                 (k r)))))))

(defn print-map
  [title annual symfn value-key legend]
  (println title)
  (println (apply str (repeat (count title) \-)))
  (doseq [line (grid->rows annual #(symfn (value-key %)))]
    (println line))
  (println legend)
  (println))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; MONTH SNAPSHOT MAPS
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn month-grid
  [dataset year month]
  (->> dataset
       (filter #(and (= (:year %) year)
                     (= (:month %) month)))
       (sort-by (juxt :row :col))
       vec))

(defn month-map-rows
  [records symfn value-key]
  (let [idx (indexed-by-cell records)]
    (for [row (reverse (range nrows))]
      (apply str
             (for [col (range ncols)]
               (let [r (get idx [row col])]
                 (symfn (value-key r))))))))

(defn print-month-map
  [title records symfn value-key legend]
  (println title)
  (println (apply str (repeat (count title) \-)))
  (doseq [line (month-map-rows records symfn value-key)]
    (println line))
  (println legend)
  (println))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; CSV EXPORT
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(defn write-monthly-csv!
  [filename dataset]
  (spit filename
        (apply str
               "year,month,row,col,lat,lon,elev,coastal,temp,rain,humidity,pressure,wind,enso,drought,flood_risk\n"
               (for [r dataset]
                 (format "%d,%d,%d,%d,%.2f,%.2f,%.2f,%s,%.2f,%.2f,%.2f,%.2f,%.2f,%.2f,%.2f,%s\n"
                         (:year r)
                         (:month r)
                         (:row r)
                         (:col r)
                         (:lat r)
                         (:lon r)
                         (:elev r)
                         (if (:coastal? r) "true" "false")
                         (:temp r)
                         (:rain r)
                         (:humidity r)
                         (:pressure r)
                         (:wind r)
                         (:enso r)
                         (:drought r)
                         (name (:flood-risk r)))))))

(defn write-annual-csv!
  [filename annual]
  (spit filename
        (apply str
               "year,row,col,lat,lon,elev,coastal,mean_temp,total_rain,mean_humidity,mean_pressure,mean_wind,mean_drought,high_flood_months,moderate_flood_months\n"
               (for [r annual]
                 (format "%d,%d,%d,%.2f,%.2f,%.2f,%s,%.2f,%.2f,%.2f,%.2f,%.2f,%.2f,%d,%d\n"
                         (:year r)
                         (:row r)
                         (:col r)
                         (:lat r)
                         (:lon r)
                         (:elev r)
                         (if (:coastal? r) "true" "false")
                         (:mean-temp r)
                         (:total-rain r)
                         (:mean-humidity r)
                         (:mean-pressure r)
                         (:mean-wind r)
                         (:mean-drought r)
                         (:high-flood-months r)
                         (:moderate-flood-months r)))))))

(defn print-sample
  [dataset n]
  (println "SAMPLE MONTHLY GRID RECORDS")
  (println "---------------------------")
  (doseq [r (take n dataset)]
    (println
     (format "y=%d m=%02d cell=(%d,%d) lat=%6.2f lon=%6.2f elev=%7.2f temp=%5.2f rain=%6.2f hum=%5.2f pres=%7.2f wind=%5.2f drought=%5.2f flood=%s"
             (:year r)
             (:month r)
             (:row r)
             (:col r)
             (:lat r)
             (:lon r)
             (:elev r)
             (:temp r)
             (:rain r)
             (:humidity r)
             (:pressure r)
             (:wind r)
             (:drought r)
             (name (:flood-risk r)))))
  (println))

(defn domain-summary
  [annual]
  {:mean-temp (round2 (average (map :mean-temp annual)))
   :mean-rain (round2 (average (map :total-rain annual)))
   :mean-humidity (round2 (average (map :mean-humidity annual)))
   :mean-pressure (round2 (average (map :mean-pressure annual)))
   :mean-wind (round2 (average (map :mean-wind annual)))})

(defn print-domain-summary
  [annual]
  (let [{:keys [mean-temp mean-rain mean-humidity mean-pressure mean-wind]}
        (domain-summary annual)]
    (println "DOMAIN SUMMARY")
    (println "--------------")
    (println (format "Mean annual temperature: %.2f C" mean-temp))
    (println (format "Mean annual rainfall:    %.2f mm" mean-rain))
    (println (format "Mean humidity:           %.2f %%" mean-humidity))
    (println (format "Mean pressure:           %.2f hPa" mean-pressure))
    (println (format "Mean wind:               %.2f km/h" mean-wind))
    (println)))

(defn print-extremes
  [annual]
  (let [hottest (apply max-key :mean-temp annual)
        wettest (apply max-key :total-rain annual)
        driest  (apply min-key :total-rain annual)
        highest (apply max-key :elev annual)]
    (println "EXTREMES")
    (println "--------")
    (println (format "Hottest cell: row=%d col=%d lat=%.2f lon=%.2f mean-temp=%.2f"
                     (:row hottest) (:col hottest) (:lat hottest) (:lon hottest) (:mean-temp hottest)))
    (println (format "Wettest cell: row=%d col=%d lat=%.2f lon=%.2f total-rain=%.2f"
                     (:row wettest) (:col wettest) (:lat wettest) (:lon wettest) (:total-rain wettest)))
    (println (format "Driest cell:  row=%d col=%d lat=%.2f lon=%.2f total-rain=%.2f"
                     (:row driest) (:col driest) (:lat driest) (:lon driest) (:total-rain driest)))
    (println (format "Highest cell: row=%d col=%d lat=%.2f lon=%.2f elev=%.2f"
                     (:row highest) (:col highest) (:lat highest) (:lon highest) (:elev highest)))
    (println)))

(defn -main
  [& _args]
  (let [dataset0 (generate-dataset 2020 2025)
        dataset  (vec (sort-by (juxt :year :month :row :col)
                               (add-drought-index dataset0)))
        annual   (annual-summaries dataset 2025)
        jan2025  (month-grid dataset 2025 1)
        jul2025  (month-grid dataset 2025 7)]

    (println "GRID-BASED SYNTHETIC CLIMATE MAP DEMO")
    (println "=====================================")
    (println)

    (print-sample dataset 12)
    (print-domain-summary annual)
    (print-extremes annual)

    (print-map
     "ANNUAL TEMPERATURE MAP (2025)"
     annual
     temp-symbol
     :mean-temp
     "A<8 B<12 C<16 D<20 E<24 F<28 G>=28 C")

    (print-map
     "ANNUAL RAINFALL MAP (2025)"
     annual
     rain-symbol
     :total-rain
     ".<200 :<400 -<700 =<1000 +<1400 *<1800 #>=1800 mm")

    (print-map
     "ANNUAL DROUGHT MAP (2025)"
     annual
     drought-symbol
     :mean-drought
     "D very dry, d dry, . near normal, w wet, W very wet")

    (print-map
     "ANNUAL FLOOD-RISK MAP (2025)"
     annual
     flood-symbol
     :high-flood-months
     "L low, M moderate, H high flood-prone")

    (print-month-map
     "JANUARY 2025 TEMPERATURE SNAPSHOT"
     jan2025
     temp-symbol
     :temp
     "A<8 B<12 C<16 D<20 E<24 F<28 G>=28 C")

    (print-month-map
     "JULY 2025 RAINFALL SNAPSHOT"
     jul2025
     rain-symbol
     :rain
     ".<20 :<40 -<70 =<100 +<140 *<180 #>=180 mm")

    (write-monthly-csv! "synthetic_climate_grid_monthly.csv" dataset)
    (write-annual-csv!  "synthetic_climate_grid_annual_2025.csv" annual)

    (println "Files written:")
    (println "  synthetic_climate_grid_monthly.csv")
    (println "  synthetic_climate_grid_annual_2025.csv")))

How to use it

Save it as:

build_and_run_climate_grid.sh

Then run:

chmod +x build_and_run_climate_grid.sh
./build_and_run_climate_grid.sh

What it produces

Inside climate-grid-demo/ it will create:

    deps.edn

    src/climate/core.clj

    synthetic_climate_grid_monthly.csv

    synthetic_climate_grid_annual_2025.csv

Manual build and run

If you want to run it later yourself:

cd climate-grid-demo
clj -M -m climate.core

Termux note

On Android Termux, first install Java and Clojure CLI if needed. A typical start is:

pkg update
pkg install openjdk-17 curl rlwrap

Then install Clojure CLI by the method supported on your Termux setup, or run the same project on Debian/UserLAnd if that is easier on your device.
Small critique of the script

This script is practical and reproducible, but it embeds the full Clojure source inline, so editing later is a bit bulky. A cleaner next step would be a two-file setup:

    one installer/build script

    one standalone core.clj

I can also give you a hardened Termux/UserLAnd installer version with dependency checks and clearer error handling.
